PAD(G) -> optional<Distances>                           call the main algorithm after decomposition
decomposeIntoSCCs(G)-> vector<Graph>

runMainAlg(G, d) -> optional<Distances>                 runs the main algorithm from Jason Li Paper
              - upper bound for weak diameter


Important Configurable Parameters:
    k = the number of rounds of BellmanDijkstra
    eps = ratio between vol(padding - U) / vol(U) = 1/alpha; In paper alpha = log(n)


PAD(Graph G) {
    Initial step: Multiply the positive edges G with 2, such that Lemma 9 from Jason Li holds

    SCC = decomposeIntoSCCs(G)

    for (auto s : SCC)
        phi = RumMainAlg(s, infinity)

    fixDagEdges(G, SCC, phi);

    computeDistances();
}

runMainAlg(G, d)                               // Input = graph of weak diameter d in G>=0

    d_approx = get a 2-aprox of diameter(G)

    d = min(d, d_approx)

    if(d small) {
        phi = runBellmanDijkstra(G, d iterations)       // If more than d negative edges => negative cycle
        return phi;
    }

    if(G small){
        phi = runBellmanDijkstra(G, n iterations)       // Maybe has cycle
        return phi;
    }

    X = runPaddedDecomposition(G, d)
    k = X.size()

    for (i = 0; i < k ; i++) {
        if( X[i] from heavy_case )                               // The only case where the weak diameter decreases
            runMainAlg(X[i], d/2)

        else {
            SCC = decomposeIntoSCCs(X[i])

            for (auto s : SCC)
                phi = RumMainAlg(s, d)

            fixDagEdges(X[i], SCC, phi);
        }

    create H as in Jason Li Paper                                  (2m + O(eps m) edges)

    phi = runBellmanDijkstra(H, k iterations)                   // If more than k iterations we have cycle: Lemma 9

    return phi;



runPaddedDecomposition(G, d)                                parameters = k, eps; Graph with non-negative edges

    Obs: To have k iterations, the padding must be d/k

Light Case:

     U_plus, U_minus = {}

     while( vol(U_plus) <= m/2 && vol(U_minus) <= m/2 )
    {
        Take s in V \ (U_plus U U_minus);

        Do for both IN and OUT:
                FOR OUT: run ot G \ U_plus
                FOR IN: run or G \ U_minus


        LIGHT CASE:

                                   d/k
                                <--------->

            s . . . . . . . . . . . | . . . . . | . . | . . . . . . . . . . . . . . . . . .
        dist=                      d/10              d/5
        vol =                                   m

        Find r <= d/10 such that:
            Ball(s, r + d/k) / Ball(s, r) <= (1 + aps) (constant) or minimum                    (Two pointers + simultaneously)
            vol(Ball(s, r)) < m
            U = U + Ball(s, r)


        OTHERWISE -> GO TO HEAVY
    }

    U = either U_plus or U_minus    (whichever has bigger volume)

    vol(U) <= 3m/2
    vol(V - U) <= 3m/2
    Vol(U + padding) <= (1 + eps) * 3m/2

    X = V - U           vol(X) <= 1.5m
    Y = U + Padding     vol(Y) <= (1 + eps) * 1.5m  => Ok if eps <= 0.33 (the smaller the better for #edges in recursion)




HEAVY CASE:
                                                   d/k
                                                <--------->

        s . . . . . . . | . . . . | . . . . . . . | . . . . . . . . . . . . . . . . . .
    dist=                        d/10            d/5
    vol =               m

    vol(Ball(s, d/10)) >= m           (This happens for both IN and OUT) directions

    Find d/10 <= r <= d/5 such that:
            Ball(s, r+d/k) / Ball(s, r) <= (1 + eps) or minimum


        X = Ball_IN(s, r+d/k) intersect Ball_OUT(s, r+d/k)

            diam of X (u -> v)
                u -> s = d/5 + d/k
                s -> v = d/5 + d/k

                u -> v < 2d/5 + 2d/k < d/2

        Y = Ball_OUT(s, r+d/k) \ B_IN(s, r)                                             <= vol(B_IN(s,r)) >= m
        Z = V \ B_OUT(s, r)                                                             <= vol(V_OUT(s, r)) >= m

        diam(X) <= d/2
        vol(Y) <= m
        vol(Z) <= m
